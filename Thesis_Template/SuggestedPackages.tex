% Unicode support 
%#######################################################################

\usepackage[utf8]{inputenc}	
\usepackage[T1]{fontenc}



% Images and figures
%#######################################################################

\usepackage{graphicx} % Include images.
\usepackage[font=footnotesize]{subcaption} % More than one image in figure environments.
\usepackage{wrapfig} % Wrap text around figures.
\usepackage{rotating} % For sideways figures/tables.
\usepackage{tikz} % Create vector graphics directly in the document.
\usetikzlibrary{positioning, calc, intersections, cd, babel} % Useful defaults.










% Tables and lists
%#######################################################################

\usepackage{colortbl} % Colors in tables.
\usepackage{longtable} % Long tables that allows page breaks.
\usepackage{multirow} % Join cells in a row.
\usepackage{paralist} % Additional counters for enumerations, e.g. [i)].









% Mathematics
%#######################################################################

\usepackage{amsmath,amsthm,amsfonts,amssymb,mathtools} % Extensive mathematics.
\usepackage{bbm} % Blackboard style of \mathbb e.g. unit operator \mathbbm{1}.
\usepackage{units} % \unit[value]{unit}. Included \nicefrac{}{}.
\usepackage{cancel} % Crossing out in formulas with \cancel{expression}.
\usepackage{resizegather} % Shrinking formula in gather to fit page.
\usepackage{tensor} %Tensor indices
\usepackage{ellipsis} %Fixes space around \ldots.
\usepackage{empheq} %Colored frames in equations.











%Other useful packages 
%#######################################################################

\usepackage{needspace} % Conditional newpage with \newpage{2cm}.
\usepackage{placeins} % \FloatBarrier forces "open" floating objects to be placed.
